DELETION IN ARRAY 
#include < iostream >
using namespace std;

int deleteElementsInRange(int arr[], int n, int l, int r)
{
	int i, newIndex = 0;
	for (i = 0; i < n; i++)
	{
		if (i <= l || i >= r)
		{
			arr[newIndex] = arr[i];
			newIndex++;
		}
	}
	return newIndex;
}
int main()
{
	int n = 9, l = 1, r = 6;
	int arr[] = {1, 2, 3, 4, 5, 6, 7, 8, 9};
	int updatedArrayLength = deleteElementsInRange(arr, n, l, r);
	for (int i = 0; i < updatedArrayLength; i++)
	{
		cout << arr[i] << " ";
	}
	cout << endl;
	return 0;
	
}

#include < iostream >
using namespace std;

class Merge
{
  private:
	int a[10], m, n, b[10], c[10];

  public:
	void accept();
	void merge();
};

void Merge::accept()
{
	int i;

	cout << "\nEnter the number of elements of 1st array: ";
	cin >> m;
	cout << "\nEnter the elements of the 1st array: ";
	for (i = 0; i < m; i++)
		cin >> a[i];
	cout << "\nEnter the number of elements of 2nd array: ";
	cin >> n;
	cout << "\nEnter the elements of the 2nd array: ";
	for (i = 0; i < n; i++)
		cin >> b[i];
}

void Merge::merge()
{
	int i, j;

	for (i = 0; i < m; i++)
		c[i] = a[i];
	j = i;
	for (i = 0; i < n; i++, j++)
		c[j] = b[i];

	printf("The merged Array");
	for (i = 0; i < m + n; i++)
		cout << c[i] << " ";
}
int main()
{
	Merge m;
	m.accept();
	m.merge();
}

int prec(char c)
{ if (c == '^')
return 3;
else if (c == '/' || c == '*')
return 2;
else if (c == '+' || c == '-')
return 1;
else
return -1; }


void infixToPostfix(string s)
{ stack< char> st;
string result;
for (int i = 0; i < s.length(); i++)
{
char c = s[i];
if ((c >= 'a' && c <= 'z') || (c >= 'A' && c <= 'Z') || (c >= '0' && c <= '9'))
result += c;
else if (c == '(')
st.push('(');
else if (c == ')')
{
while (st.top() != '(')
{
result += st.top();
st.pop();
}
st.pop();
}
else
{
while (!st.empty() && prec(s[i]) <= prec(st.top()))

st.pop();
}
}
st.push(c);
}
}
while (!st.empty())
{
result += st.top();
