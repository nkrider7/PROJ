// C++ object Program
#include < iostream >
using namespace std;
class A
{
public :
void show(){
cout<<"Welcome to ds college";
}
};
int main()
{
class A obj;
obj.show();
return 0;
}

OUTPUT : Welcome to ds college
// C++ Sum Program
#include < iostream >
using namespace std;
class A
{
int a , b , sum ;
public :
void get(){
cout<<"Enter your number";
cin>>a>>b;
sum = a + b;
}
void show(){ cout<< sum; } };
int main()
{
class A obj;
obj.get(); obj.show();
return 0;
}
